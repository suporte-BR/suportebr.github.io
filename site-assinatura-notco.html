    <script>

        // --- "Banco de Dados" de Imagens ---

        // Adicione ou remova os links das imagens que você quer oferecer.

        const imageDatabase = [

            'https://placehold.co/80x80/6366f1/ffffff?text=L1',

            'https://placehold.co/80x80/ec4899/ffffff?text=L2',

            'https://placehold.co/80x80/22c55e/ffffff?text=L3',

            'https://placehold.co/80x80/f97316/ffffff?text=L4',

            'https://placehold.co/80x80/14b8a6/ffffff?text=L5'

        ];


        const gallery = document.getElementById('logo-gallery');

        const previewLogo = document.getElementById('preview-logo');


        // Popula a galeria com as imagens do "banco de dados"

        imageDatabase.forEach((url, index) => {

            const img = document.createElement('img');

            img.src = url;

            img.alt = `Logo ${index + 1}`;

            img.classList.add('gallery-logo', 'w-16', 'h-16');

            

            // Seleciona a primeira imagem por padrão

            if (index === 0) {

                img.classList.add('selected');

                previewLogo.src = url;

            }


            gallery.appendChild(img);

        });


        // Adiciona um listener de clique na galeria

        gallery.addEventListener('click', function(event) {

            // Verifica se o clique foi em uma imagem da galeria

            if (event.target.tagName === 'IMG') {

                // Remove a seleção de todas as outras imagens

                gallery.querySelectorAll('.gallery-logo').forEach(img => {

                    img.classList.remove('selected');

                });

                

                // Adiciona a classe 'selected' à imagem clicada

                const selectedImg = event.target;

                selectedImg.classList.add('selected');

                

                // Atualiza a imagem na pré-visualização

                previewLogo.src = selectedImg.src;

            }

        });


        // Função para atualizar a pré-visualização da assinatura em tempo real

        function updateSignature() {

            const nome = document.getElementById('nome').value || 'Seu Nome';

            const email = document.getElementById('email').value || 'seu.email@exemplo.com';

            const telefone = document.getElementById('telefone').value || '+55 (00) 00000-0000';

            const cargo = document.getElementById('cargo').value || 'Seu Cargo';

            const pais = document.getElementById('pais').value || 'Seu País';


            document.getElementById('preview-nome').innerText = nome;

            document.getElementById('preview-email').innerText = email;

            document.getElementById('preview-telefone').innerText = telefone;

            document.getElementById('preview-cargo').innerText = cargo;

            document.getElementById('preview-pais').innerText = pais;

        }


        // Adiciona um listener ao botão de download

        document.getElementById('downloadBtn').addEventListener('click', function() {

            const signatureElement = document.getElementById('signature-preview');

            const downloadButton = this;


            downloadButton.innerText = 'Gerando imagem...';

            downloadButton.disabled = true;


            html2canvas(signatureElement, {

                backgroundColor: null, // Fundo transparente

                scale: 2, // Aumenta a escala para melhor qualidade de imagem

                useCORS: true // Necessário para carregar imagens de outros domínios no canvas

            }).then(canvas => {

                const link = document.createElement('a');

                link.download = 'assinatura.png';

                link.href = canvas.toDataURL('image/png');

                link.click();

                downloadButton.innerText = 'Fazer Download da Assinatura (PNG)';

                downloadButton.disabled = false;

            }).catch(err => {

                console.error('Ocorreu um erro ao gerar a imagem:', err);

                alert('Não foi possível gerar a imagem. Tente novamente.');

                downloadButton.innerText = 'Fazer Download da Assinatura (PNG)';

                downloadButton.disabled = false;

            });

        });


        // Chama a função uma vez no início para garantir que a pré-visualização não esteja vazia

        updateSignature();

    </script>
